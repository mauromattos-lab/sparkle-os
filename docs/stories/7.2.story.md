### Schema completo

-- Tenants
CREATE TABLE IF NOT EXISTS zenya_tenants (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  system_prompt TEXT NOT NULL,
  active_tools JSONB NOT NULL DEFAULT '[]',
  chatwoot_account_id TEXT NOT NULL UNIQUE,

### Closure pattern — CRÍTICO para segurança
```typescript
// CORRETO — tenantId por closure, invisível ao LLM
export function createTenantTools(tenantId: string, config: TenantConfig) {
  return {
    escalarHumano: tool({
      description: 'Escala o atendimento para um humano',
      parameters: z.object({
        resumo_conversa: z.string(),
        ultima_mensagem: z.string(),
      }),
      execute: async ({ resumo_conversa, ultima_mensagem }) => {
        // tenantId vem do closure — o LLM nunca tem acesso a ele
        return await escalateToHuman(tenantId, config, { resumo_conversa, ultima_mensagem });
      }
    }),
    // ... outras tools
  };
}

// ERRADO — tenantId como parâmetro (LLM pode manipular)
// parameters: z.object({ tenantId: z.string(), ... })
```

### Cache do TenantConfigLoader
```typescript
const cache = new Map<string, { config: TenantConfig; expiresAt: number }>();
const TTL_MS = 5 * 60 * 1000; // 5 minutos

async function loadTenantConfig(tenantId: string): Promise<TenantConfig> {
  const cached = cache.get(tenantId);
  if (cached && cached.expiresAt > Date.now()) return cached.config;
  const config = await db.query('SELECT * FROM zenya_tenants WHERE id = $1', [tenantId]);
  cache.set(tenantId, { config: config.rows[0], expiresAt: Date.now() + TTL_MS });
  return config.rows[0];
}
```

### Criptografia AES-256-GCM
```typescript
import { createCipheriv, createDecipheriv, randomBytes } from 'node:crypto';

export function encryptCredential(value: string, masterKey: string): Buffer {
  const iv = randomBytes(16);
  const cipher = createCipheriv('aes-256-gcm', Buffer.from(masterKey, 'hex'), iv);
  const encrypted = Buffer.concat([cipher.update(value, 'utf8'), cipher.final()]);
  const authTag = cipher.getAuthTag();
  return Buffer.concat([iv, authTag, encrypted]);
}
```

### Identificar tenant_id a partir do webhook
O `chatwoot_account_id` vem do webhook (`body.account.id`). O TenantConfigLoader busca o tenant via `SELECT id FROM zenya_tenants WHERE chatwoot_account_id = $1`. Este lookup também deve ser cacheado.
