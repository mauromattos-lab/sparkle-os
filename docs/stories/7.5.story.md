## Dev Notes

### Google Calendar API — autenticação OAuth2
```typescript
import { google } from 'googleapis';

function getCalendarClient(credentials: GoogleCalendarCredentials) {
  const auth = new google.auth.OAuth2(
    credentials.client_id,
    credentials.client_secret,
  );
  auth.setCredentials({ refresh_token: credentials.refresh_token });
  return google.calendar({ version: 'v3', auth });
}
```

### Tool: buscarJanelasDisponiveis
```typescript
buscarJanelasDisponiveis: tool({
  description: 'Busca horários disponíveis para agendamento',
  parameters: z.object({
    data_inicio: z.string().describe('Data início no formato ISO 8601'),
    data_fim: z.string().describe('Data fim no formato ISO 8601'),
  }),
  execute: async ({ data_inicio, data_fim }) => {
    const calendar = getCalendarClient(await getDecryptedCredentials(tenantId, 'google_calendar'));
    const response = await calendar.freebusy.query({
      requestBody: {
        timeMin: data_inicio,
        timeMax: data_fim,
        items: [{ id: config.calendar_id }],
      },
    });
    // Retorna slots livres calculados
  }
})
```

### Ativação condicional por active_tools
```typescript
// Em createTenantTools():
const tools: Record<string, Tool> = {
  escalarHumano: ...,
  enviarTextoSeparado: ...,
};

if (config.active_tools.includes('google_calendar')) {
  Object.assign(tools, createCalendarTools(tenantId, config));
}
```

### Duração padrão de agendamento
O n8n usava `agendamento_duracao_minutos: 30` como config. Adicionar `calendar_duration_minutes` (padrão 30) na tabela `zenya_tenants` ou no JSON de `active_tools`.
